
create table member( 	
 	idx number(15) primary key,
 	nickName varchar2(20),
 	email varchar2(50),
 	division number(2)
);

create table store(
	nickName_idx number(15) not null,
	name varchar2(20),
	address1 varchar2(50),
	address2 varchar2(50),
	tel varchar2(50),
	openClose varchar2(50),
	photo1 varchar2(50),
 	photo2 varchar2(50),
 	info CLOB,
 	good number(10)
);  

CREATE sequence surgery_seq
 
create table surgery (
	surgery_idx number(15) primary key, 
	nickName_idx number(15), 
	category number(5),
	name varchar2(50),
	price number(20),
	photo varchar2(50)
);


CREATE sequence staff_seq

create table staff (
	staff_idx number(10) primary key,
	nickName_idx number(15),
	name varchar2(50),
	info varchar2(50),
    grade varchar2(50),
	photo varchar2(50)
);


create table good (
	login_idx number(15), 
	store_idx number(15)
);
 
CREATE sequence reservation_seq 

create table reservation (
	reservation_idx number(10) primary key,
	login_idx number(10),
	store_idx number(10),
	staff_idx number(10),
	cal_day varchar2(50),
	getTime varchar2(50),
	surgery_name varchar2(50),
	price number(10),
	complete number(5),
	regdate date 
);

create table mileage (
	login_idx number(10),
	user_point number(10)
);  
	 
CREATE sequence review_seq

create table review (
	review_idx number(10) primary key,
	login_idx number(10), 
	reservation_idx number(10),
	store_idx number(10),
	staff_name varchar2(50), 
	context varchar2(500),
	rating number(10), 
	regdate date,
	complete number(5) 
);

create table reply (
	review_idx number(10), 
	staff_name varchar2(50),
	context varchar2(100)
);

create table product (
	login_idx number(10),
	surgery_idx number(10),
	regdate date  
);
